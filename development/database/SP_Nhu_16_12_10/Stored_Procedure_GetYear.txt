USE `tmdt`;
DROP procedure IF EXISTS `Giao_Dich_GetYear`;

DELIMITER $$
USE `tmdt`$$
CREATE DEFINER=`root`@`localhost` PROCEDURE `Giao_Dich_GetYear`()
BEGIN
SELECT DISTINCT YEAR(ngay_tao) AS sort_year FROM giao_dich;
END$$

DELIMITER ;
